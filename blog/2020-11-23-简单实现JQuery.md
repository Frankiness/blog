## JQuery本质上是遍历元素......
<!-- truncate -->
```javascript
class JQuery{
  constructor(selector){
    this.arr = []
    const result = document.querySelectorAll(selector) //类数组
    for(let i=0;i<result.length;i++){		//	遍历将类数组中的value放入数组
      this.arr[i] = result[i]
    }
    this.selector = selector
  }
  //get方法
  get(index){
    return this.arr[index]
  }
  //each方法(参数是回调函数)
  each(Fn){
    for(let item of this.arr){	//遍历数组将每个元素放入回调函数中执行
      const elem = item
      Fn(elem)
    }
  }
  //on事件监听(参数是事件类型，回调函数)
  on(type,Fn){
    return this.arr.forEach(item => {		//遍历给每个标签添加事件和回调函数
      item.addEventListener(type,Fn,false)
    });
  }
}
```
